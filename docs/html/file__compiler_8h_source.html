39// Context for managing compilation of source GLSL files into destination
40// SPIR-V files or preprocessed output.
41class FileCompiler {
42 public:
43 enum class SpirvBinaryEmissionFormat {
44 Unspecified, // No binary output format specified, this is the only valid
45 // option when the compilation output is not in SPIR-V binary
46 // code form.
47 Binary, // Emits SPIR-V binary code directly.
48 Numbers, // Emits SPIR-V binary code as a list of hex numbers.
49 CInitList, // Emits SPIR-V binary code as a C-style initializer list
50 // of hex numbers.
51 WGSL, // Emits SPIR-V module converted to WGSL source text.
52 // Requires a build with Tint support.
53 };
54
55 FileCompiler()
56 : output_type_(OutputType::SpirvBinary),
57 binary_emission_format_(SpirvBinaryEmissionFormat::Unspecified),
58 needs_linking_(true),
59 total_warnings_(0),
60 total_errors_(0) {}

62 // Compiles a shader received as specified by input_file, returning true
63 // on success and false otherwise. If force_shader_stage is not
64 // shaderc_glsl_infer_source or any default shader stage then the given
65 // shader_stage will be used, otherwise it will be determined from the source
66 // or the file type.
67 //
68 // Places the compilation output into a new file whose name is derived from
69 // input_file according to the rules from glslc/README.asciidoc.
70 //
71 // If version/profile has been forced, the shader's version/profile is set to
72 // that value regardless of the #version directive in the source code.
73 //
74 // Any errors/warnings found in the shader source will be output to std::cerr
75 // and increment the counts reported by OutputMessages().
76 bool CompileShaderFile(const InputFileSpec& input_file);
77
78 // Adds a directory to be searched when processing #include directives.
79 //
80 // Best practice: if you add an empty string before any other path, that will
81 // correctly resolve both absolute paths and paths relative to the current
82 // working directory.
83 void AddIncludeDirectory(const std::string& path);
84
85 // Sets the output filename. A name of "-" indicates standard output.
86 void SetOutputFileName(const shaderc_util::string_piece& file) {
87 output_file_name_ = file;
88 }
89
90 // Sets the format for SPIR-V binary compilation output.
91 void SetSpirvBinaryOutputFormat(SpirvBinaryEmissionFormat format) {
92 binary_emission_format_ = format;

95 // Returns false if any options are incompatible. The num_files parameter
96 // represents the number of files that will be compiled.
97 bool ValidateOptions(size_t num_files);

99 // Outputs to std::cerr the number of warnings and errors if there are any.
100 void OutputMessages();

102 // Sets the flag to indicate individual compilation mode. In this mode, all
103 // files are compiled individually and written to separate output files
104 // instead of linked together. This method also disables linking and sets the
105 // output file extension to ".spv". Disassembly mode and preprocessing only
106 // mode override this mode and flags.
107 void SetIndividualCompilationFlag();
108
109 // Sets the flag to indicate disassembly mode. In this mode, the compiler
110 // emits disassembled textual output, instead of outputting object files.
111 // This method also sets the output file extension to ".spvasm" and disables
112 // linking. This mode overrides individual compilation mode, and preprocessing
113 // only mode overrides this mode.
114 void SetDisassemblyFlag();
115
116 // Sets the flag to indicate preprocessing only mode. In this mode, instead of
117 // outputting object files, the compiler emits the preprocessed source files.
118 // This method disables linking and sets the output file to stdout. This mode
119 // overrides disassembly mode and individual compilation mode.
120 void SetPreprocessingOnlyFlag();

122 // Gets the reference of the compiler options which reflects the command-line

126 // Gets a pointer which points to the dependency info dumping hander. Creates

137 enum class OutputType {
138 SpirvBinary, // A binary module, as defined by the SPIR-V specification.
139 SpirvAssemblyText, // Assembly syntax defined by the SPIRV-Tools project.
140 PreprocessedText, // Preprocessed source code.
141 };

143 // Emits the compilation output from the given result to the given output
144 // file and returns true if the result represents a successful compilation
145 // step. Otherwise returns false, possibly emits messages to the standard
146 // error stream, and does not produce an output file. Accumulates error
147 // and warning counts for use by the OutputMessages() method.
148 template <typename CompilationResultType>
149 bool EmitCompiledResult(
150 const CompilationResultType& result, const std::string& input_file_name,
151 const std::string& output_file_name,
152 shaderc_util::string_piece error_file_name,
153 const std::unordered_set<std::string>& used_source_files);
154
155 // Returns the final file name to be used for the output file.
156 //
157 // If an output file name is specified by the SetOutputFileName(), use that
158 // argument as the final output file name.
159 //
160 // If the user did not specify an output filename:
161 // If linking is not required, and the input filename has a
162 // standard stage extension (e.g. .vert) then returns the input filename
163 // without directory names but with the result extenstion (e.g. .spv or
164 // .spvasm) appended.
165 //
166 // If linking is not required, and the input file name does not have a
167 // standard stage extension, then also returns the directory-stripped input
168 // filename, but replaces its extension with the result extension. (If the
169 // resolved input filename does not have an extension, then appends the
170 // result extension.)
171 //
172 // If linking is required and output filename is not specified, returns
173 // "a.spv".
174 std::string GetOutputFileName(std::string input_filename);
175
176 // Returns the candidate output file name deduced from input file name and
177 // user specified output file name. It is computed as follows:
178 //
179 // If the user did specify an output filename and the compiler is not in
180 // preprocessing-only mode, then returns that file name.
181 //
182 // If the user did not specify an output filename:
183 // If the input filename has a standard stage extension (e.g. .vert) then
184 // returns the input filename without directory names but with the result
185 // extenstion (e.g. .spv or .spvasm) appended.
186 //
187 // If the input file name does not have a standard stage extension, then also
188 // returns the directory-stripped input filename, but replaces its extension
189 // with the result extension. (If the resolved input filename does not have
190 // an extension, then appends the result extension.)
191 //
192 // When a resolved extension is not available because the compiler is in
193 // preprocessing-only mode or the compilation requires linking, use .spv as
194 // the extension.
195 std::string GetCandidateOutputFileName(std::string input_filename);
196
197 // Returns true if the compiler's output is preprocessed text.
198 bool PreprocessingOnly() {
199 return output_type_ == OutputType::PreprocessedText;
200 }

233 // Counts errors encountered in all compilations via this object.
234 size_t total_errors_;
